Function to filter out all the keys of a flat object that have objects or arrays using Object.keys

let obj = {
    a:"Apple",
    b:["Basketball","Baseball"],
    c: {
     call: "cellphone"
    },
    d: "Dog"
}

function filterObject(obj)
{
    let ans = {}
    Object.keys(obj).forEach(i=>{
        if(typeof(obj[i])!=typeof({})) 
          ans[i]=obj[i]
    })
    return ans
}

let ans = filterObject(obj)
console.log(ans)


Output :
{a: 'Apple', d: 'Dog'}



Function to filter out all the keys of a flat object that have objects or arrays using Object.entries

let obj = {
    a:"Apple",
    b:["Basketball","Baseball"],
    c: {
     call: "cellphone"
    },
    d: "Dog"
}

function filterObject(obj)
{
    let ans = {}
    Object.entries(obj).forEach(arr=>{
        if(typeof(arr[1])!=typeof({})) 
          ans[arr[0]]=arr[1]
    })
    return ans
}

let ans = filterObject(obj)
console.log(ans)

Output : 
{a: 'Apple', d: 'Dog'}
