//App.js

import React,{ useState } from "react";
function App()
{
 //declaring a new state variable name count
  const [count , setCount] = useState(0);
  return(
  <>
    <p>count value is : {count}</p>
    <button onClick={()=> setCount(count+1)}>update</button>
  </>);
}

export default App;

//Here we have created a state count with initial value 0 which will keep on updating as we click the update button.
